'Создать таск' : 'Обновить таск' f.input_hidden(:id) f.input_text('Название таска', :name, params.merge({:class => 'input-xxlarge'})) f.select('Категория', :category, params.merge({:values => @categories, :selected => (@task.category.nil? ? nil : @task.category.id) })) f.textarea('Описание таска', :description, params.merge({:class => 'input-xxlarge', :rows => 8})) f.input_text('Ответ', :answer_regex, params.merge({:class => 'input-xxlarge'})) f.input_text('Стоимость', :price, params.merge({:class => 'input-mini'})) cb_opts = {:show_value => false} cb_opts[:disabled] = true unless logged_admin? f.input_checkbox("Опубликован", :is_published, checked=(@task.is_published ? @task.is_published : nil), cb_opts) f.input_submit(submit_label, {:class => 'btn btn-primary'}) f.input_hidden(:csrf_token, @csrf_token) end} #{if @submit_action == :update form_for(@task, :method => :post, :action => TaskController.r(:delete)) do |f| f.input_hidden(:id) f.input_submit('Удалить таск', {:class => 'btn btn-danger', :onClick => 'return confirm("А уверен ли?")'}) f.input_hidden(:csrf_token, @csrf_token) end end}
